// Add this to your existing script in index.html function loadDynamicContent() { const saved = localStorage.getItem('progettoAdminData'); if(saved) { const data = JSON.parse(saved); // Apply colors if(data.colors) { document.documentElement.style.setProperty('--primary-color', data.colors.primary); document.body.style.backgroundColor = data.colors.bg; document.body.style.color = data.colors.text; } // Apply fonts if(data.fonts) { document.body.style.fontFamily = data.fonts.body; document.body.style.fontSize = data.fonts.fontSize + 'px'; document.querySelectorAll('h1, h2, h3, .section-title, .hero-content h1').forEach(el => { el.style.fontFamily = data.fonts.heading; }); } // Display projects if(data.projects && data.projects.length > 0) { const projectsContainer = document.getElementById('projectsGrid'); if(projectsContainer) { projectsContainer.innerHTML = data.projects.map(proj => `

${proj.title}

${proj.desc}

`).join(''); } } // Update hero text if(data.texts) { const heroTitle = document.querySelector('[data-key="heroTitle"]'); if(heroTitle) heroTitle.innerHTML = data.texts.heroTitle; const heroSubtitle = document.querySelector('[data-key="heroSubtitle"]'); if(heroSubtitle) heroSubtitle.textContent = data.texts.heroSubtitle; const aboutText = document.getElementById('aboutTextContainer'); if(aboutText && data.texts.aboutText) aboutText.innerHTML = data.texts.aboutText; } } } // Call this after your existing setLanguage function loadDynamicContent(); // Listen for changes from admin panel window.addEventListener('storage', function(e) { if(e.key === 'progettoAdminData') { location.reload(); } });
